def fetch_youtube_data(api_key, query="AI", max_results=10):
    """YouTube Data API v3ë¡ ê²ì"""
    import requests
    url = "https://www.googleapis.com/youtube/v3/search"
    params = {
        "part": "snippet", "q": query,
        "type": "video", "maxResults": max_results,
        "order": "viewCount", "regionCode": "KR",

    }
    r = requests.get(url, params=params)
    return r.json()
